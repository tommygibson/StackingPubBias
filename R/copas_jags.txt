model
{
  for(i in 1:S){
    y[i] ~ dnorm(mean.y[i], prec.y[i])
    
    mean.y[i] = theta + rho * s[i] * (z[i] - (gamma0 + gamma1 / s[i]))
    prec.y[i] = 1 / var.y[i]
    var.y[i] = tau2 + s2[i] * (1 - (rho ^ 2))
    
    z[i] ~ dnorm(gamma0 + gamma1 / s[i], 1) T(0,)
    
    s2[i] = pow(s[i], 2)
    
  }
  
  tau2 = pow(tau, 2)
  rho ~ dunif(-.99, .99)
  #rho = rho_t * 2 - 1
  
  tau ~ dt(0, 1, 1) T(0,)
  # rho_t ~ dbeta(2, 2)
  theta ~ dnorm(0, 0.01)
  
}
